########################################################################
# Setup library
########################################################################
include(GrPlatform) #define LIB_SUFFIX

include_directories(${Boost_INCLUDE_DIR})
link_directories(${Boost_LIBRARY_DIRS})

list(APPEND fm0_sources
)

set(fm0_sources "${fm0_sources}" PARENT_SCOPE)
if(NOT fm0_sources)
	MESSAGE(STATUS "No C++ sources... skipping lib/")
	return()
endif(NOT fm0_sources)

add_library(gnuradio-fm0 SHARED ${fm0_sources})
target_link_libraries(gnuradio-fm0 ${Boost_LIBRARIES} ${GNURADIO_ALL_LIBRARIES})
set_target_properties(gnuradio-fm0 PROPERTIES DEFINE_SYMBOL "gnuradio_fm0_EXPORTS")

if(APPLE)
    set_target_properties(gnuradio-fm0 PROPERTIES
        INSTALL_NAME_DIR "${CMAKE_INSTALL_PREFIX}/lib"
    )
endif(APPLE)

########################################################################
# Install built library files
########################################################################
include(GrMiscUtils)
GR_LIBRARY_FOO(gnuradio-fm0 RUNTIME_COMPONENT "fm0_runtime" DEVEL_COMPONENT "fm0_devel")

########################################################################
# Build and register unit test
########################################################################
include(GrTest)

include_directories(${CPPUNIT_INCLUDE_DIRS})

list(APPEND test_fm0_sources
    ${CMAKE_CURRENT_SOURCE_DIR}/test_fm0.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_fm0.cc
)

add_executable(test-fm0 ${test_fm0_sources})

target_link_libraries(
  test-fm0
  ${GNURADIO_RUNTIME_LIBRARIES}
  ${Boost_LIBRARIES}
  ${CPPUNIT_LIBRARIES}
  gnuradio-fm0
)

GR_ADD_TEST(test_fm0 test-fm0)

########################################################################
# Print summary
########################################################################
message(STATUS "Using install prefix: ${CMAKE_INSTALL_PREFIX}")
message(STATUS "Building for version: ${VERSION} / ${LIBVER}")
